libconfig

app.cfg

version="1.0";
app = {
    a=1;
    b=2.0;
    c=[1,2,3];
    d=(1,"2",true);
    e={
        f=-1;
    };
};

code

#include <iostream>
#include "libconfig.h++"

int main(int, char**) {
    libconfig::Config cfg;
    try{
        cfg.readFile("app.cfg");
        int a = cfg.lookup("app.a");
        double b = cfg.lookup("app.b");
        const auto& root = cfg.getRoot();
        for(const int& n:root["app"]["c"])
        {
            std::cout << n << "\n";
        }
        for(const libconfig::Setting& item:root["app"]["d"])
        {
            auto type = item.getType();
            switch (type)
            {
            case libconfig::Setting::Type::TypeInt:
                std::cout << int(item);
                break;
            case libconfig::Setting::Type::TypeString:
                std::cout << std::string(item);
                break;
            case libconfig::Setting::Type::TypeBoolean:
                std::cout << bool(item);
                break;
            }
            std::cout << "\n";
        }
        int f = cfg.lookup("app.e.f");
        std::cout << f << "\n";
    }
    catch(const std::exception& e)
    {
        std::cout << e.what();
        exit(-1);
    }
    return 0;
}
